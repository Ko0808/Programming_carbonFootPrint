--- 距離計算 --- async def get_coords_from_api(place_name): """ Nominatim APIを直接呼び出して地名から座標を取得する """ query = place_name.replace(" ", "+") url = f"https://nominatim.openstreetmap.org/search?q={query}&format=jsonv2&limit=1" try: response = await window.fetch(url, method="GET") if not response.ok: return None data = (await response.json()).to_py() if data: return (float(data[0]['lat']), float(data[0]['lon'])) return None except Exception as e: print(f"API request failed: {e}") return None def haversine_distance(coords1, coords2): """ 2つの座標間の距離を計算する（ハヴァサイン公式） """ R = 6371 # 地球の半径 (km) lat1, lon1 = coords1 lat2, lon2 = coords2 phi1, phi2 = math.radians(lat1), math.radians(lat2) dphi = math.radians(lat2 - lat1) dlambda = math.radians(lon2 - lon1) a = math.sin(dphi / 2)**2 + \ math.cos(phi1) * math.cos(phi2) * math.sin(dlambda / 2)**2 return 2 * R * math.atan2(math.sqrt(a), math.sqrt(1 - a)) # この関数を追加してください async def get_distance(origin, destination): """ 2つの地名から距離を計算する """ try: # 非同期で両方の座標を取得する coords1_task = get_coords_from_api(origin) coords2_task = get_coords_from_api(destination) coords1, coords2 = await asyncio.gather(coords1_task, coords2_task) # 両方の座標が正常に取得できた場合のみ、距離を計算 if coords1 and coords2: return haversine_distance(coords1, coords2) else: missing_places = [] if not coords1: missing_places.append(origin) if not coords2: missing_places.append(destination) alert(f"地名を特定できませんでした: {', '.join(missing_places)}") return None except Exception as e: print(f"Distance calculation error: {e}") alert("距離の計算中に予期せぬエラーが発生しました。") return None # --- 計算ロジック --- async def calculate_daily_summary(event): global daily_entries if not daily_entries: alert("今日のリストに食材を追加してください。") return show_loading(True) profile = json.loads(window.localStorage.getItem('user_profile')) user_residence = profile['residence'] summary_data = [] total_weight = 0 total_cfp = 0 total_domestic_cfp = 0 for entry in daily_entries: food_info = food_database[entry['food_id']] # 重量換算 weight = entry['quantity'] if entry['unit'] == '個': if food_info['avg_weight_kg']: weight *= food_info['avg_weight_kg'] else: alert(f"{food_info['name_jp']}は重量(kg)で入力してください。") show_loading(False) return # 距離計算 distance = await get_distance(entry['origin'], user_residence) domestic_distance = await get_distance(food_info['domestic_origin'], user_residence) if distance is None or domestic_distance is None: show_loading(False) return # 距離計算失敗 # CFP計算 cfp = weight * distance * EMISSION_FACTOR_PER_KM domestic_cfp = weight * domestic_distance * EMISSION_FACTOR_PER_KM summary_data.append({ "name": food_info['name_jp'], "weight": weight, "cfp": cfp, "domestic_cfp": domestic_cfp }) total_weight += weight total_cfp += cfp total_domestic_cfp += domestic_cfp # 結果をテーブルに表示 summary_table_body = document.querySelector("#summary-table tbody") summary_table_body.innerHTML = "" for item in summary_data: row = document.createElement("tr") row.innerHTML = f"{item['name']}{item['weight']:.2f}{item['cfp']:.2f}{item['domestic_cfp']:.2f}" summary_table_body.appendChild(row) document.getElementById("total-weight").innerText = f"{total_weight:.2f}" document.getElementById("total-cfp").innerText = f"{total_cfp:.2f}" document.getElementById("total-domestic-cfp").innerText = f"{total_domestic_cfp:.2f}" document.getElementById("summary-section").style.display = "block" # 累計データに保存 all_records_str = window.localStorage.getItem('all_records') all_records = json.loads(all_records_str) if all_records_str else [] all_records.append({ "date": datetime.now().strftime('%Y-%m-%d'), "total_cfp": total_cfp, "total_domestic_cfp": total_domestic_cfp }) window.localStorage.setItem('all_records', json.dumps(all_records)) #
